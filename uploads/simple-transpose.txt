#include <stdio.h>

void accept(int input[10][10], int rows, int cols) {

	for (int i = 0; i < rows; i++) {

		for (int j = 0; j < cols; j++) {
			printf("Enter matrix element %d%d ", j, i);
			scanf_s("%d", &input[i][j]);

		}

	}


}

void print(int input[10][10], int rows, int cols) {

	for (int i = 0; i < rows; i++){
		for (int j = 0; j < cols; j++) {
			printf("%d ", input[i][j]);
		}
		printf("\n");
	}

}
void print_compact(int input[10][3], int rows) {

	for (int i = 0; i < rows; i++) {
		for (int j = 0; j < 3; j++) {
			printf("%d ", input[i][j]);
		}
		printf("\n");
	}

}



void compaction(int input[10][10], int rows, int cols)
{
	int output[10][3];
	int k =1;

	for (int i = 0; i < rows; i++) {

		for (int j = 0; j < cols; j++) {

			if (input[i][j] != 0) {

				output[k][0] = i;
				output[k][1] = j;
				output[k][2] = input[i][j];
				k++;


			}


		}

	}

	output[0][0] = rows;
	output[0][1] = cols;
	output[0][2] = k - 1;

	print_compact(output, output[0][2]+1);
	printf("output simple transpose:--------------------\n");
	simple_transpose(output, rows);

}

void simple_transpose(int input[10][3], int rows) {

	int output[10][3];
	int k = 1;
	int num2chk = 0;
	int temp;


	for (num2chk = 0; num2chk < input[0][1]; num2chk++) {

		for (int i = 1; i < rows; i++) {

			if (input[i][1] == num2chk) {

				output[k][0] = input[i][1];
				output[k][1] = input[i][0];
				output[k][2] = input[i][2];
				k++;
			}
		}
		output[0][0] = input[0][1];
		output[0][1] = input[0][0];
		output[0][2] = input[0][2];

	}
	print_compact(output, output[0][2] + 1);


}



int main() {

	int x[100][100],rows,cols;

	printf("enter rows and cols of matrix: \n");
	scanf_s("%d %d", &rows, &cols);
	accept(x, rows, cols);
	printf("Matrix:--------------------\n");
	print(x, rows, cols);
	printf("output:--------------------\n");
	compaction(x, rows, cols);
	

	return 0;
}
