#include <stdio.h>
#include <stdlib.h>

void accept (int a[10][10] , int c, int d)
{
    for (int i =0;i <c ; i++)
    {
        for (int j =0 ; j<d ; j++)
        {
            scanf("%d",&a[i][j]);
            
        }
    }
}

void print (int a[10][10],int c,int d)
{
    for (int i =0; i<c;i++)
    {
        for (int j =0 ; j< d;j++)
        {
            printf( "%d " , a[i][j]);
        }
        printf("\n");
    }
}

void add (int x[10][10] , int y [10][10] ,int a, int b)
{
    printf ("addition:\n");
    
    int c [10][10] ;
    
    for (int i =0 ; i<a ;i++){
        
        for (int j =0; j<b;j++)
        {
            c[i][j] = x[i][j] + y[i][j];
        }
    }
    
    print (c,a,b);
}

void sub (int x[10][10] , int y [10][10] ,int a, int b)
{
    printf ("sub:\n");
    
    int c [10][10] ;
    
    for (int i =0 ; i<a ;i++){
        
        for (int j =0; j<b;j++)
        {
            c[i][j] = x[i][j] - y[i][j];
        }
    }
    
    print (c,a,b);
}

void mul (int x[10][10] , int y [10][10] ,int a, int b, int m ,int n)
{
    if (b != m)
    {
        printf ("Error: Columns of matrix 1 must match rows of matrix 2 for multiplication!\n");
        return;
    }
    
    printf ("mul:\n");
    
    int c [10][10] ;
    
    for (int i = 0 ; i < a ; i++) {
        for (int j = 0; j < n; j++) {
            c[i][j] = 0;
            for (int k = 0; k < b; k++) {
                c[i][j] += x[i][k] * y[k][j];
            }
        }
    }
    
    print (c, a, n);
}

void trn (int x[10][10] ,int a, int b)
{
    printf ("transpose:\n");
    
    int c [10][10] ;
    
    for (int i =0 ; i<a ;i++){
        
        for (int j =0; j<b;j++)
        {
            c[j][i] = x[i][j];
        }
    }
    
    print (c,b,a);
}

int main (){
    
    int x[10][10] , a [10][10] ;
    int y,z, m2y,m2z;
    
    printf("enter matrix 1 length (rows and columns): ");
    scanf ("%d %d",&y ,&z);
    
    if (y <= 0 || y > 10 || z <= 0 || z > 10) 
    {
        printf ("Invalid input dimensions! Must be between 1 and 10.\n"); 
        return 0;
    }
    
    printf ("enter matrix 1 data:\n");
    accept(x,y,z);
       
    printf("enter matrix 2 length (rows and columns): ");
    scanf ("%d %d",&m2y ,&m2z);
    
    if (m2y <= 0 || m2y > 10 || m2z <= 0 || m2z > 10) 
    {
        printf ("Invalid input dimensions! Must be between 1 and 10.\n"); 
        return 0;
    }
    
    printf ("enter matrix 2 data:\n");
    accept(a,m2y,m2z);
    
    printf ("\nmatrix 1 data:\n");
    print(x,y,z);
 
    printf ("\nmatrix 2 data:\n");
    print(a,m2y,m2z);
    while (1){
    printf ("\nchoose: \n1)add \n2)subtract \n3)multiply \n4)transpose\n5)exit");
    int input;
    scanf ("%d",&input);
    
    switch (input) {
        
        case 1: 
            if (y != m2y || z != m2z) {
                printf("Error: Matrices must have identical dimensions for addition!\n");
            } else {
                add (x,a,y,z);
            }
            break;
        case 2: 
            if (y != m2y || z != m2z) {
                printf("Error: Matrices must have identical dimensions for subtraction!\n");
            } else {
                sub (x,a,y,z);
            }
            break;
        case 3: 
            mul (x,a,y,z,m2y,m2z);
            break;
        case 4: 
            trn (x,y,z);
            break;
         case 5: 
         exit(0);
        default: 
            printf("invalid input\n");
            
    }
    }
    return 0;
}
